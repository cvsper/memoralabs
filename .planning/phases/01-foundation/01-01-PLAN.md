    - `usage_log` table: id INTEGER PK AUTOINCREMENT, tenant_id TEXT NOT NULL, operation TEXT NOT NULL, endpoint TEXT NOT NULL, status_code INTEGER NOT NULL, latency_ms INTEGER, tokens_used INTEGER DEFAULT 0, created_at INTEGER NOT NULL, FK tenant_id -> tenants(id)
